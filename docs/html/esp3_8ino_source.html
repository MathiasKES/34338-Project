193void playPattern(const unsigned int* timings, uint8_t length) {
194
195 // Prevent overlapping sound patterns
196 if (buzzerState != BUZZER_IDLE) return;
197
198 currentPattern.timings = timings;
199 currentPattern.length = length;
200
201 stepIndex = 0;
202 buzzerState = BUZZER_ON;
203 lastChange = millis();
204
205 digitalWrite(BUZZER_PIN, HIGH); // Start with buzzer ON
206}
207
213void updateBuzzer() {
214 if (buzzerState == BUZZER_IDLE) return;
215
216 unsigned long now = millis();
217
218 // Wait until current timing interval expires
219 if (now - lastChange < currentPattern.timings[stepIndex]) return;
220
221 lastChange = now;
222 stepIndex++;
223
224 // End of pattern reached
225 if (stepIndex >= currentPattern.length) {
226 digitalWrite(BUZZER_PIN, LOW);
227 buzzerState = BUZZER_IDLE;
228 return;
229 }
230
231 // Toggle buzzer state
232 if (buzzerState == BUZZER_ON) {
233 digitalWrite(BUZZER_PIN, LOW);
234 buzzerState = BUZZER_OFF;
235 } else {
236 digitalWrite(BUZZER_PIN, HIGH);
237 buzzerState = BUZZER_ON;
238 }
239}
240
244void stopBuzzer() {
245 buzzerState = BUZZER_IDLE;
246 digitalWrite(BUZZER_PIN, LOW);
247}

304
318void callback(char* topic, byte* payload, unsigned int length) {
319
320 // Ignore empty MQTT messages
321 if (length == 0) return;
322
323 StaticJsonDocument<512> doc;
324 DeserializationError err = deserializeJson(doc, payload, length);
325
326 // Abort if JSON parsing fails
327 if (err) {
328 Serial.print("JSON parse failed: ");
329 Serial.println(err.c_str());
330 return;
331 }
332
333 // ---------------------------------------------------------------------------
334 // Keypad PIN verification response
335 // ---------------------------------------------------------------------------
336 if (strcmp(topic, net.makeTopic("access/keypad_response").c_str()) == 0) {
337
338 // Ignore keypad responses during admin servo control
339 if (adminServoControl) return;
340
341 accessGranted = (doc["response"]["accessGranted"] | false)
342 ? AccessResult::Granted
343 : AccessResult::Denied;
344
345 if (accessGranted != AccessResult::Granted) {
346 Serial.println("Access Denied");
347 playDeniedSound();
348 forceLock();
349 return;
350 }
351
352 // Access granted: unlock door
353 Serial.println("Unlocking door");
354 playUnlockSound();
355
356 lock_servo.write(180); // Move servo to unlocked position
357 servoOpen = true;
358
359 digitalWrite(GREEN_PIN, HIGH);
360 digitalWrite(RED_PIN, LOW);
361
362 unlocked = true;
363 unlockUntil = millis() + UNLOCK_TIME_MS;
364 }
365
366 // ---------------------------------------------------------------------------
367 // RFID access response (only react to denial)
368 // ---------------------------------------------------------------------------
369 else if (strcmp(topic, net.makeTopic("access/response").c_str()) == 0) {
370
371 rfidAccess = (doc["response"]["hasAccess"] | false)
372 ? true
373 : false;
374
375 if (!rfidAccess) {
376 Serial.println("Access Denied");
377 playDeniedSound();
378 forceLock();
379 return;
380 }
381 }
382
383 // ---------------------------------------------------------------------------
384 // Keypad tap feedback
385 // ---------------------------------------------------------------------------
386 else if (strcmp(topic, net.makeTopic("keypad/beep").c_str()) == 0) {
387
388 // Only beep if RFID access is valid
389 if (!rfidAccess) return;
390
391 playTapSound();
392 }
393
394 // ---------------------------------------------------------------------------
395 // Admin servo control enable/disable
396 // ---------------------------------------------------------------------------
397 else if (strcmp(topic, net.makeTopic("admin/servo_control").c_str()) == 0) {
398
399 adminServoControl = (doc["data"]["adminServoControl"] | false)
400 ? true
401 : false;
402
403 if (!adminServoControl) {
404 Serial.println("Admin servo control disabled");
405 forceLock(); // Reset servo to locked state
406 return;
407 }
408
409 Serial.println("Admin servo control enabled");
410 }
411}

484void loop() {
485 net.loop();
486 yield();
487
488 // Advance buzzer state machine
489 updateBuzzer();
490
491 // ---------------------------------------------------------------------------
492 // Admin servo control mode
493 // ---------------------------------------------------------------------------
494 if (adminServoControl) {
495
496 // Map potentiometer value to servo angle
497 servoAngle = (int)(analogRead(POT_PIN) / 1023.0f * 180.0f);
498 servoAngle = constrain(servoAngle, 0, 180);
499
500 lock_servo.write(servoAngle);
501 return;
502 }
503
504 // ---------------------------------------------------------------------------
505 // Automatic relock after unlock timeout
506 // ---------------------------------------------------------------------------
507 const uint32_t now = millis();
508
509 if (unlocked && (int32_t)(now - unlockUntil) >= 0) {
510 Serial.println("Locking door");
511 playLockSound();
512 forceLock();
513 }
514}
