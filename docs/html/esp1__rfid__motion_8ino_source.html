221void makeEnteredPins(uint8_t pinLength)
222{
223 // Clamp PIN length to display capacity (4 digits)
224 if (pinLength > 4) {
225 pinLength = 4;
226 }
227
228 // Fill entered portion with masking characters
229 memset(enteredPins, '*', pinLength);
230
231 // Fill remaining positions with spaces to clear old characters
232 memset(enteredPins + pinLength, ' ', 4 - pinLength);
233
234 // Ensure string is null-terminated
235 enteredPins[4] = '\0';
236}

250void callback(char* topic, byte* payload, unsigned int length) {
251 // Ignore empty MQTT messages
252 if (length == 0) return;
253
254 StaticJsonDocument<512> doc;
255
256 // Deserialize JSON payload into document
257 DeserializationError err = deserializeJson(doc, payload, length);
258
259 // Abort if JSON parsing fails
260 if (err) {
261 Serial.print("JSON parse failed: ");
262 Serial.println(err.c_str());
263 return;
264 }
265
266 // ---------------------------------------------------------------------------
267 // RFID access response
268 // ---------------------------------------------------------------------------
269 if (strcmp(topic, net.makeTopic("access/response").c_str()) == 0) {
270
271 // Timestamp when request was sent (provided by backend)
272 uint32_t requestMs = doc["sent_ts_ms"] | 0;
273
274 // Compute round-trip delay to detect stale responses
275 uint32_t deltaMs = millis() - requestMs;
276
277 Serial.printf(
278 "Request/Response time diff: %lu ms\n",
279 deltaMs
280 );
281
282 // Ignore responses that took too long to arrive
283 if (deltaMs > DISPLAY_MS) {
284 Serial.println("Took too long to respond");
285 return;
286 }
287
288 // Extract access decision from JSON payload
289 rfidAccess = (doc["response"]["hasAccess"] | false)
290 ? AccessResult::Granted
291 : AccessResult::Denied;
292
293 Serial.println("UID match: waiting for PIN...");
294
295 // Handle denied RFID access immediately
296 if (rfidAccess != AccessResult::Granted) {
297 lcdPrintLine(F("Access Denied"), 0);
298 textshown = true;
299 showTextUntil = millis() + DISPLAY_MS;
300 return;
301 }
302
303 // Prompt user to enter PIN after successful RFID
304 lcdPrintLine(F("Enter PIN:"), 0);
305 textshown = true;
306 showTextUntil = millis() + PIN_TIME_MS;
307
308 }
309 // ---------------------------------------------------------------------------
310 // Keypad PIN verification response
311 // ---------------------------------------------------------------------------
312 else if (strcmp(topic, net.makeTopic("access/keypad_response").c_str()) == 0) {
313
314 // Ignore PIN responses if RFID was not authorized
315 if (rfidAccess != AccessResult::Granted) return;
316
317 // Extract PIN validation result
318 accessGranted = (doc["response"]["accessGranted"] | false)
319 ? AccessResult::Granted
320 : AccessResult::Denied;
321
322 // Handle incorrect PIN
323 if (accessGranted != AccessResult::Granted) {
324 Serial.println("Access Denied");
325 lcdPrintLine(F("Access Denied"), 0);
326 textshown = true;
327 showTextUntil = millis() + DISPLAY_MS;
328 return;
329 }
330
331 // Handle successful authentication
332 Serial.println("Access Granted");
333 lcdPrintLine(F("Access Granted"), 0);
334 textshown = true;
335 showTextUntil = millis() + UNLOCK_TIME_MS;
336
337 }
338 // ---------------------------------------------------------------------------
339 // Keypad visual feedback (keypress)
340 // ---------------------------------------------------------------------------
341 else if (strcmp(topic, net.makeTopic("keypad/beep").c_str()) == 0) {
342
343 // Only show keypad feedback after successful RFID
344 if (rfidAccess != AccessResult::Granted) return;
345
346 // Number of digits entered so far
347 uint8_t pinLength = doc["data"]["pinlength"] | 0;
348
349 // Update masked PIN display
350 makeEnteredPins(pinLength);
351 lcdPrintLine(enteredPins, 1);
352 }
353}

374void onMotionDetected(uint32_t now) {

426void uidToHexString(const MFRC522::Uid& uid, char* output, size_t outputSize) {
427
428 // Ensure output buffer is large enough:
429 // 2 hex characters per byte + null terminator
430 if (outputSize < (uid.size * 2 + 1)) {
431 output[0] = '\0'; // Fail safely with empty string
432 return;
433 }
434
435 // Convert each UID byte into two hex characters
436 for (byte i = 0; i < uid.size; i++) {
437 sprintf(&output[i * 2], "%02X", uid.uidByte[i]);
438 }
439
440 // Explicitly terminate C-string
441 output[uid.size * 2] = '\0';
442}

450void handleRFID() {
451
452 // Exit if no new card is present
453 if (!mfrc522.PICC_IsNewCardPresent()) {
454 return;
455 }
456
457 // Exit if card UID could not be read
458 if (!mfrc522.PICC_ReadCardSerial()) {
459 return;
460 }
461
462 char uidString[21];
463 uidToHexString(mfrc522.uid, uidString, sizeof(uidString));
464
465 Serial.println(uidString);
466 lcdPrintLine(F("Connecting..."), 0);
467
468 textshown = true;
469 showTextUntil = millis() + DISPLAY_MS;
470
471 // Build JSON payload for access request
472 StaticJsonDocument<64> data;
473 data["uid"] = uidString;
474 data["event"] = "RFID_Try";
475
476 // Publish access request via MQTT
477 bool ok = net.publishJson("access/request", data);
478 Serial.println(ok ? "MQTT publish OK" : "MQTT publish FAILED");
479
480 // Properly halt card communication
481 mfrc522.PICC_HaltA();
482 mfrc522.PCD_StopCrypto1();
483}

546void loop() {
547 net.loop(); // Process MQTT traffic
548 yield(); // Allow background WiFi tasks
549
550 const uint32_t now = millis();
551
552 // If a message is currently displayed, wait for timeout
553 if (textshown) {
554 if ((int32_t)(now - showTextUntil) >= 0) {
555 forceLock(); // Reset system when timeout expires
556 }
557 delay(POLL_MS);
558 return;
559 }
560
561 // Update LCD backlight based on motion sensor
562 updateMotionState(now);
563
564 // Only allow RFID scans when display is active
565 if (isDisplayActive(now)) {
566 handleRFID();
567 }
568}
