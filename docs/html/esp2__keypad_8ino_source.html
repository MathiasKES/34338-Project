138
150void callback(char* topic, byte* payload, unsigned int length) {
151
152 // Ignore empty MQTT messages
153 if (length == 0) return;
154
155 StaticJsonDocument<512> doc;
156
157 // Parse JSON payload
158 DeserializationError err = deserializeJson(doc, payload, length);
159
160 // Abort on JSON parsing failure
161 if (err) {
162 Serial.print("JSON parse failed: ");
163 Serial.println(err.c_str());
164 return;
165 }
166
167 // ---------------------------------------------------------------------------
168 // RFID access response: enable or disable keypad
169 // ---------------------------------------------------------------------------
170 if (strcmp(topic, net.makeTopic("access/response").c_str()) == 0) {
171
172 // Enable keypad only if RFID access was granted
173 kpEnabled = (doc["response"]["hasAccess"] | false)
174 ? true
175 : false;
176 }
177 // ---------------------------------------------------------------------------
178 // PIN verification response: always disable keypad afterward
179 // ---------------------------------------------------------------------------
180 else if (strcmp(topic, net.makeTopic("access/keypad_response").c_str()) == 0) {
181
182 // Prevent further input until next RFID authorization
183 kpEnabled = false;
184 }
185}

265void loop() {
266 net.loop(); // Process MQTT traffic
267 yield(); // Allow background WiFi tasks
268
269 // Read keypad state (non-blocking)
270 char key = keypad.getKey();
271
272 // Ignore input if no key pressed or keypad is disabled
273 if (key == NO_KEY || !kpEnabled) {
274 return;
275 }
276
277 // ---------------------------------------------------------------------------
278 // Numeric key input
279 // ---------------------------------------------------------------------------
280 if (key >= '0' && key <= '9') {
281
282 // Only accept input if buffer is not full
283 if (currentIdx < CODE_LENGTH) {
284 input[currentIdx++] = key; // Store digit
285 input[currentIdx] = '\0'; // Maintain null-terminated string
286
287 Serial.print("Key: ");
288 Serial.println(key);
289 }
290 }
291
292 // ---------------------------------------------------------------------------
293 // Submit PIN using '#'
294 // ---------------------------------------------------------------------------
295 else if (key == '#') {
296
297 // Only submit if required PIN length is reached
298 if (currentIdx == CODE_LENGTH) {
299 Serial.print("Submitting code: ");
300 Serial.println(input);
301
302 // Build JSON payload for PIN submission
303 StaticJsonDocument<64> data;
304 data["event"] = "KP_try";
305 data["code"] = input;
306
307 net.publishJson("keypad/submit", data);
308 }
309 else {
310 Serial.println("Code too short, resetting.");
311 }
312
313 // Always reset buffer after submission attempt
314 reset_code();
315 }
316
317 // ---------------------------------------------------------------------------
318 // Clear input using '*'
319 // ---------------------------------------------------------------------------
320 else if (key == '*') {
321 reset_code();
322 Serial.println("Input cleared");
323 }
324
325 // Ignore unsupported keys (A-D)
326 else {
327 return;
328 }
329
330 // Any valid keypress is treated as a tap event
331 publishTap();
332}
